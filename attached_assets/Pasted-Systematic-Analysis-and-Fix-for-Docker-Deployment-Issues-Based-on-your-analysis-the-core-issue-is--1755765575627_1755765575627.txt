Step 3: Create a Systematic Dockerfile

Let's build a Dockerfile that handles both development and production scenarios:
bash

cat > Dockerfile << 'EOF'
FROM node:20-alpine3.19

# Install minimal build dependencies
RUN apk add --no-cache python3 make g++ git

# Create user (but we'll build as root first, then switch)
RUN addgroup -g 1001 -S runner && adduser -u 1001 -S runner -G runner

# Set working directory
WORKDIR /app

# Copy package files
COPY package*.json ./

# Install all dependencies (including dev dependencies for build)
RUN npm ci --no-audit --prefer-offline

# Copy source code
COPY . .

# Set production environment
ENV NODE_ENV=production

# Build application
RUN npm run build

# Check what the build produced
RUN find . -name "*.js" -o -name "*.ts" | grep -E "(index|main|server|app)" | head -10 || echo "No obvious entry point found"

# Now switch to non-root user for security
RUN chown -R runner:runner /app
USER runner

# Determine the correct startup command
CMD ["sh", "-c", "
  # First try: if dist/index.js exists, use it
  if [ -f 'dist/index.js' ]; then
    echo 'Starting from dist/index.js'
    exec node dist/index.js
  # Second try: if server/index.js exists, use it  
  elif [ -f 'server/index.js' ]; then
    echo 'Starting from server/index.js'
    exec node server/index.js
  # Third try: if there's a built version somewhere
  elif find . -name '*.js' | grep -q 'index.js'; then
    ENTRY=$(find . -name 'index.js' | head -1)
    echo 'Starting from found index.js at $ENTRY'
    exec node \"$ENTRY\"
  # Fallback: use npm start
  else
    echo 'Falling back to npm start'
    exec npm start
  fi
"]
EOF

Step 5: Alternative Approach - Development Mode

If the production build isn't working, let's try a development-mode container:
bash

cat > Dockerfile.dev << 'EOF'
FROM node:20-alpine3.19

# Install minimal build dependencies including tsx
RUN apk add --no-cache python3 make g++ git

# Create user
RUN addgroup -g 1001 -S runner && adduser -u 1001 -S runner -G runner

# Set working directory
WORKDIR /app

# Copy package files
COPY package*.json ./

# Install all dependencies
RUN npm ci --no-audit --prefer-offline

# Copy source code
COPY . .

# Switch to non-root user
RUN chown -R runner:runner /app
USER runner

# Run in development mode

# Build and test development version

  -p 3001:5000 \
  --name addypin-dev \

  -e DATABASE_URL="postgresql://addypin:your_db_password@localhost:5432/addypin" \

Step 6: Fix Nginx Configuration
bash

# Update nginx to handle both dev and prod
cat > /etc/nginx/conf.d/addypin.conf << 'EOF'
server {
    listen 80;
    server_name addypin.com www.addypin.com 155.94.144.191;
    
    # Proxy to production container
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        
        # Increase timeouts
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }
}

# Optional: dev server on different port
server {
    listen 3001;
    server_name addypin.com www.addypin.com 155.94.144.191;
    
    location / {
        proxy_pass http://localhost:3001;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
EOF

# Test and reload nginx
nginx -t
systemctl reload nginx
