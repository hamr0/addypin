cd /opt/addypin

# Stop current service

systemctl stop addypin

# Backup current app and switch to new version

if [ -d app ]; then
  mv app app-backup-$DEPLOY_TIME
fi
mv app-new-$DEPLOY_TIME app

# Update environment variables and permissions

  echo "✅ Service is running"

  systemctl stop addypin
