Solution

Let's create a proper Dockerfile that works with your project structure:
bash

cd /opt/addypin/addypin-repo

# Create a corrected Dockerfile
cat > Dockerfile << 'EOF'
# Use exact Node.js version from Replit
FROM node:20.19.3-bullseye-slim

# Install build tools
RUN apt-get update && apt-get install -y \
    gcc \
    g++ \
    make \
    python3 \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Update npm
RUN npm install -g npm@11.5.2

# Create non-root user
RUN groupadd -r runner && useradd -r -g runner runner

# Set working directory
WORKDIR /home/runner/workspace
RUN chown -R runner:runner /home/runner/workspace

# Fix npm cache permissions
RUN mkdir -p /home/runner/.npm && chown -R runner:runner /home/runner/.npm
RUN npm config set cache /home/runner/.npm --global

# Copy package files first for better caching
COPY --chown=runner:runner package*.json ./

# Switch to runner user
USER runner

# Install ALL dependencies
RUN npm ci

# Copy ALL source files (from the current directory which is addypin-repo)
COPY --chown=runner:runner . .

# Debug: Check what files we have
RUN echo "=== Current directory structure ==="
RUN ls -la
RUN echo "=== Client directory ==="
RUN ls -la client/
RUN echo "=== Check if index.html exists ==="
RUN find . -name "index.html" -type f

# Build the application
RUN npm run build

# Remove dev dependencies after build
RUN npm prune --omit=dev

# Set production environment
ENV NODE_ENV=production

# Expose the port
EXPOSE 5000

# Start the application
CMD ["npm", "start"]
EOF

# Now build from the correct directory
docker build -t addypin-app .
